class GfG
{
   
    public static int getNth(Node node, int ind)
    {
       //Your code here
      
       Node current_node = node;
       int idx=0;
     for(int i=1; i<=ind; i++){
         if(ind==i){
            idx =current_node.data;
         }
         
         current_node = current_node.next;
     }
     
     return idx;
    }
}
